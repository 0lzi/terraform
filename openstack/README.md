  OS_APPLICATION_CREDENTIAL_ID_CREDENTIAL_ID
  OS_APPLICATION_CREDENTIAL_SECRETN_CREDENTIAL_SECRET

  AWS_ACCESS_KEY_IDY_ID
  AWS_SECRET_ACCESS_KEYCESS_KEY
  AWS_S3_ENDPOINTNT
  AWS_REGION
